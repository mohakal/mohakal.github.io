        function Employee() {
          var name;
          var age;
          var salary;
  
          function getName() {
            return name;
          }
  
          function getAge() {
            return age;
          }
  
          function getSalary() {
            return salary;
          }
  
          function setName(n) {
            name = n;
          }
  
          function setAge(a) {
            age = a;
          }
  
          function setSalary(s) {
            salary = s;
          }
  
          function increaseSalary(percentage) {
            setSalary(getSalary() + getSalary() * percentage / 100);
          }
  
          function incrementAge() {
            setAge(getAge() + 1);
          }
  
          return {
            setName: setName,
            setAge: setAge,
            setSalary: setSalary,
            increaseSalary: increaseSalary,
            incrementAge: incrementAge
          }
        }
